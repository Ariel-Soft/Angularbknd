Dynamic Grid Example

ngback-ng-grid Settings

All settings can be provided as attributes in the ngback-ng-grid or through the query string parameters.

  • view-name (Defaults: null) : A name of a table or view from the database

  • filter-options (Defaults: null) : uses BackAnd REST API filter options: {fieldName, operator, value}

  • input-style (Defaults: null) : css style

ngback-ng-grid Events

  • gridConfigCompleted Emit when the configuration of the grid finish loading from the BackAnd REST API and before the configuration of the ng-grid

  • gridDataCompleted Emit when the data of the grid finish loading from the BackAnd REST API and before the loading of the rows into ng-grid

  • setCellTemplate Emit when setting the cellTemplate option of ng-grid

  • Markup
  • JavaScript 
<div>
   Static Grid Example
   <ngback-ng-grid view-name="'Employees'" input-style="{'height': 300}"></ngback-ng-grid>
</div>
<div data-ng-controller="dynamicGridController">
   Dynamic Grid Example
   <ngback-ng-grid view-name="viewName" input-style="inputStyle"></ngback-ng-grid>
</div>
'use strict';
var backAndControllers  = angular.module('backAnd.controllers');
backAndControllers.controller('dynamicGridController', ['$scope', function ($scope)  {
    $scope.viewName  = 'Customers';
    $scope.inputStyle  =  {
        'height': 300
    };
}
])

ngback-form Settings

All settings can be provided as attributes in the ngback-ng-grid or through the query string parameters.

  • view-name (Defaults: null) : A name of a table or view from the database

  • row-id (Defaults: null) : The row primary key, if provided then the form is in EDIT mode otherwise it is in a CREATE mode. if the primary key of the row has multiple column values then they should be comma delimited in the primary key columns' order

  • default-fields-values (Defaults: null) : Default value for each field: {fieldName, value}

  • Markup
  • JavaScript 
<div>
   Static Form Example
   <div ngback-form view-name="'Employees'" row-id="1"></div>
</div>
<div data-ng-controller="dynamicFormController">
   Dynamic Form Example
   <div ngback-form view-name="viewName" row-id="id"></div>
</div>
'use strict';
var backAndControllers  = angular.module('backAnd.controllers');
backAndControllers.controller('dynamicFormController', ['$scope', function ($scope)  {
    $scope.viewName  = 'Customers';
    $scope.id  = 1;
}
])
